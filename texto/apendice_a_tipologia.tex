\appendix{Apêndice A: Tipologias para Ontologias}
\label{apendice:a-tipologia}

\section{Quanto à função Mizoguchi; Vanwelkenbuyse n; Ikeda (1995)}

\begin{itemize}
    \item \textbf{Ontologias de domínio:} reutilizáveis no domínio, fornecem vocabulário sobre conceitos e seus relacionamentos, sobre as atividades e regras que os governam;
    \item \textbf{Ontologias de tarefa:} fornecem um vocabulário sistematizado de termos, por meio da especificação de tarefas que podem ou não estar no mesmo domínio;
    \item \textbf{Ontologias gerais:} incluem um vocabulário relacionado a coisas, eventos, tempo, espaço, casualidade, comportamento, funções etc.
\end{itemize}


\section{Quanto ao grau de formalismo Uschold; Gruninger (1996)}

\begin{itemize}
    \item \textbf{Ontologias altamente informais:} expressa livremente em linguagem natural;
    \item \textbf{Ontologias semiinformais:} expressa em linguagem natural de forma restrita e estruturada;
    \item \textbf{Ontologias semiformais:} expressa em uma linguagem artificial definida formalment;
    \item \textbf{Ontologias rigorosamente formais:} os termos são definidos com semântica formal, teoremas e provas.
\end{itemize}


\section{Quanto à aplicação Jasper; Uschold (1999)}

\begin{itemize}
    \item \textbf{Ontologias de autoria neutra:} uma ontologia escrita em uma única língua e depois convertida para uso em diversos sistemas, reutilizando-se as informações;
    \item \textbf{Ontologias como especificação:} cria-se uma ontologia para um domínio, a qual é usada para documentação e manutenção no desenvolvimento de softwares;
    \item \textbf{Ontologias de acesso comum à informação:} quando o vocabulário é inacessível, a ontologia torna inteligível a informação, proporcionando conhecimento compartilhado dos termo.
\end{itemize}


\section{Quanto à estrutura Haav; Lubi (2001)}

\begin{itemize}
    \item \textbf{Ontologias de alto nível:} descrevem conceitos gerais relacionados a todos os elementos da ontologia (espaço, tempo, matéria, objeto, evento etc.), os quais são independentes do problema ou domínio;
    \item \textbf{Ontologias de domínio:} descrevem o vocabulário relacionado a um domínio, como, por exemplo, medicina ou automóveis;
    \item \textbf{Ontologias de tarefa:} descrevem uma tarefa ou atividade, como por exemplo, diagnósticos ou compras, mediante inserção de termos especializados na ontologia.
\end{itemize}


\section{Quanto ao conteúdo Van-Heijist; Schreiber; Wielinga (2002)}

\begin{itemize}
    \item \textbf{Ontologias de informação:} Especificam a estrutura de registros de bancos de dados (por exemplo, os esquemas de bancos de dados);
    \item \textbf{Ontologias terminológicas:} Especificam termos que serão usados para representar o conhecimento em um domínio (por exemplo, os léxicos);
    \item \textbf{Ontologias de modelagem do conhecimento:} Especificam conceituações do conhecimento, têm uma estrutura interna semanticamente rica e são refinadas para uso no domínio do conhecimento que descreve;
    \item \textbf{Ontologias de aplicação:} Contêm as definições necessárias para modelar o conhecimento em uma aplicação;
    \item \textbf{Ontologias de domínio:} Expressam conceituações que são específicas para um determinado domínio do conhecimento;
    \item \textbf{Ontologias genéricas:} Similares às ontologias de domínio, mas os conceitos que as definem são considerados genéricos e comuns a vários campo;
    \item \textbf{Ontologias de representação:} Explicam as conceituações que estão por trás dos formalismos de representação do conhecimento.
\end{itemize}
